gs -dNOPAUSE -sDEVICE=pdfwrite -sOUTPUTFILE=combinedpdf.pdf -dBATCH 1.pdf 2.pdf 3.pdf
alternative: 

pdftk *pdf cat output ../FINAL_NAME.pdf
extract images from pdf 

pdfimages -j Arco\ Big\ Walls-en.pdf  arco-big
